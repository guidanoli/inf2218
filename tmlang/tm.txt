program := symbols_def tape_def state_def_list
symbols_def := SYMBOLS symbol_list
symbol_list := symbol_list SYMBOL | SYMBOL
tape_def := TAPES tape_name_list
tape_name_list := tape_name_list TAPE_NAME | TAPE_NAME
state_def_list := state_def_list state_def | state_def
state_def := WHEN STATE_NAME DO stmt END
stmt := pass_stmt | seq_stmt | ifelse_stmt | write_stmt | move_stmt | goto_stmt
pass_stmt := PASS
seq_stmt := stmt stmt
ifelse_stmt := IF cond THEN stmt ELSE stmt END
write_stmt := TAPE_NAME ASSIGN_ARROW exp 
move_stmt := TAPE_NAME direction
goto_stmt := GOTO STATE_NAME
direction := LEFT | RIGHT
cond := eq_cond
eq_cond := exp EQ exp
exp := literal_exp | read_exp
literal_exp := SYMBOL
read_exp := TAPE_NAME
